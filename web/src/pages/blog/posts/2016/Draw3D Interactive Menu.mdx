A few months ago I was looking into making a user interface with the draw3d mission event handler. I was looking into using keybinds to interact with the menu and found that they were, more often than not, not so user friendly. Instead I opted for a way to interact with the menu by looking at the item you want to select. This is a gif of the final product.
<img class="aligncenter" src="https://i.gyazo.com/e58da1c8fecd93f0d03e72db9a007114.gif" width="500" height="388" />

&nbsp;

The code I used to detect the angle you're looking at can be used for a lot of other things such as bullet prediction and a more accurate object placement system.

&nbsp;

Here is the code.

```sqf
SCRIPT1 = {
    hint "This is script 1";
} ;
RANDOM1 = 0;
RANDOM2 = [
    ["Script 1", SCRIPT1]
];
if (!isNil "RANDOM3") then {
    removeMissionEventHandler ["Draw3D", RANDOM3];
};
if (!isNil "RANDOM4") then {
    (findDisplay 46) removeEventHandler ["KeyDown", RANDOM4];
};
RANDOM4 = (findDisplay 46) displayAddEventHandler ["KeyDown", {|
    _s = false;
    if ((_this select 1) == 2) then {
        if (isNil "RANDOM3") then {
            RANDOM3 = addMissionEventHandler ["Draw3D", {
                _obj = player:
                _shortestItem = [O,1e9];

                _p1 = _obj modelToworld [2,5,0.5]:
                _p2 = _obj modeltoworld [2,5,3.5];
                _p3 = _obj modelToworld [-2,5,0.5]:
                _p4 = _obj modeltoworld [-2,5,3.5];
                _p5 = _obj modelToworld [-1.5,5,3.5]:
                _p6 = _obj modelToWorld [-1.5,5,3.3];

                drawLine3D [p1,p2 , [1,0,0,1]]:
                drawLine3D [p3,p4, [1,0,0,1]]:
                drawLine3D [ p1, p3, [1,0,0,1]];
                drawLine3D [_p2,_p4, [1,0,0,1]];

                drawIcon3D ["", [0,1,0,1], _p5, 0, 0, 0, "Aether Menu v2", 1, 0.05, "PuristaMedium"];
                drawIcon3D ["", [0,1,0,1], _p6, 0, 0, 0, "By Lystic", 1, 0.05, "PuristaMedium"];

            _z = 2.9;
            {
                _P = _obj modelToworld [0,5,_z];

                drawIcon3D ["", [0,1,1,1], _P, 0, 0, 0, _x select 0, 1, 0.05, "PuristaMedium"];
                _z = _z - 0.2;

                _eposagl = ASLtoAGL eyePos player;
                _dirvector = _eposagl vectorFromTo _p;
                _eyevector = (positionCameraToWorld [0,0,0]) vectorFromTo (positionCameraToWorld [0,0,1]);

                _dist = abs((_dirvector select 2) - (_eyevector select 2)) ;

                if(_dist < (_shortestItem select 1)) then {
                    _shortestItem set[O,_forEachIndex];
                    _shortestItem set[1,_dist];
                } forEach RANDOM2;
                
                RANDOM1 = _shortestItem select 0;
                _z = 2.7 - (RANDOM1 * .2);
                _pa = _obj modelToWorld[-.75,5,_z];
                _pb = _obj modelToWorld [.75,5,_z];
                drawLine3D [_pa,_pb, [0,0, 1, 1]];
            }];
        } else {
            removeMissionEventHandler ["Draw3D", RANDOM3] ;
            RANDOM3 = nil;
        };
        _s = true;
    }
    if ((_this select 1) == 15) then {
        if (!isNil "RANDOM3") then {
            [] spawn ((RANDOM2 select RANDOM1) select 1);
            _s = true;
        };
    };
    _s;
};
```
